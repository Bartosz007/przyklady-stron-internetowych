atrybuty:

disabled="disabled" - wyłączenie

size="liczba" - liczba widocznych znaków
checked="checked"- wartość domyślna(pola wyboru)
src="źródło pliku"
placeholder="znak wodny"
required=""required" - pole wymagane

max="liczba" - wartość maksymalna
accept="plik/rozszerzenie" -akceptacja plików
